# Checks if gender laws/culture/religion allows someone to fight
warrior_trigger = {
	OR = {
		AND = {
			is_female = yes
			OR = {
				religion = gaian
				religion = ursuline
			}
		}
		
		has_character_flag = special_marshal
		
		has_law = true_cognatic_succession
		AND = {
			has_law = agnatic_succession
			is_female = no
		}
		AND = {
			has_law = cognatic_succession
			OR = {
				is_female = no
				martial = 20
			}
		}
		AND = {
			has_law = enatic_succession
			is_female = yes
		}
		AND = {
			has_law = enatic_cognatic_succession
			OR = {
				is_female = yes
				martial = 20
			}
		}
		
		trait = adventurer
		trait = viking
		trait = duelist
		trait = berserker
		trait = peasant_leader
	}
}

	OR = {
		AND = {
			society_member_of = secret_religious_society_xxxx

		}

	OR = {
		AND = {
			has_character_flag = character_was_xxxx

	    	}
	    }
	}
}
